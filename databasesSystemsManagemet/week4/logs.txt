CREATE TABLE accounts_log (
    logDate DATE,
    logType VARCHAR (6),
    logId NUMBER);

CREATE OR REPLACE TRIGGER accounts_access
    BEFORE INSERT OR UPDATE OR DELETE
    ON accounts
    FOR EACH ROW
DECLARE
    log_type accounts_log.logType%type;
    log_custid NUMBER;
BEGIN
    IF DELETING THEN
    log_type := 'Delete';
    log_custid := old.custid;
    ELSIF UPDATING THEN
    log_type := 'Update';
    log_custid := old.custid;
    ELSIF INSERTING THEN
    log_type := 'Insert';
    log_custid := new.custid;
    END IF;
    
    INSERT INTO accounts_log (logDate, logType, logId)
        VALUES(SYSDATE, log_type, log_custid);
END;
