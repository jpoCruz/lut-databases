set serveroutput on format wrapped;

CREATE OR REPLACE PROCEDURE passwordChecker(
    user_id IN NUMBER
    user_password IN VARCHAR)
BEGIN
    SELECT custid, passwrd
    FROM CUSM
    WHERE custid = user_id
    
    IF user_password == passwrd THEN
        UPDATE CUSM SET SignOnFlag = 'Y' WHERE custid = user_id;
        DBMS_OUTPUT.put_line('true password');
    ELSE
        UPDATE CUSM SET SignOnFlag = 'N' WHERE custid = user_id;
        DBMS_OUTPUT.put_line('false password');
    END IF;
END passwordChecker;
